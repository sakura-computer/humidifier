def initialize
  try_require_sdk('aws-sdk-v1')
  try_require_sdk('aws-sdk')

  self.shim =
    if Object.const_defined?(:AWS)
      AwsAdapters::SDKV1.new
    elsif Object.const_defined?(:Aws)
      AwsAdapters::SDKV2.new
    else
      AwsAdapters::Noop.new
    end
end
