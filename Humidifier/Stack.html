def initialize(opts = {})
  self.name = opts[:name]
  self.id   = opts[:id]

  ENUMERABLE_RESOURCES.values.each { |prop| send(:#{prop}=", opts.fetch(prop, {})) }
  STATIC_RESOURCES.values.each { |prop| send(:#{prop}=", opts[prop]) }
end
